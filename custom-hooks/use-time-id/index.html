еҪ“еңЁеӣһи°ғдёӯдҪҝз”ЁвҖӢ

иҝҷж ·жҳҜжӣҙзӣҙжҺҘзҡ„пјҢдҪҶд№ҹе№¶дёҚжҳҜдёҖж— жҳҜеӨ„гҖӮеҰӮжһңйңҖиҰҒеңЁеӨ–йғЁеҮҪж•°дёӯдҪҝз”Ёж—¶пјҢеҸҜиғҪе°ұдёҚиғҪе°Ҷ timeId ж”ҫеҲ° sueEffect еҶ…йғЁпјҢиҖҢжҳҜдҪҝз”Ё useRef иҝӣиЎҢеӮЁеӯҳпјҢе№¶еңЁеҗҲйҖӮзҡ„ж—¶еҖҷиҝӣиЎҢй”ҖжҜҒгҖӮ

import { useRef } from 'react';

export default function Home() {
  const timeId = useRef<NodeJS.Timeout |null>(null);

  function doSome () {
    timeId.current = setTimeout(() => {
      ...
    }, 1011)
  }

  useEffect(() => {
    // зңӢпјҢиҝҷж—¶еҖҷжҳҜдёҚжҳҜе°ұеҫ—еҚ•зӢ¬еӨ„зҗҶ `timeId`
    return () =>{
      if( timeId.current )
        clearTimeout(timeId.current);}
  },[]);

  return <div onclick={ doSome }>...</div>
}

import { useRef, useEffect } from 'react';

export function useTimeId() {
  const timeId = useRef<NodeJS.Timeout>(undefined));

  useEffect(
    () =>
        ()=>
            timeId.current &&
              clearTimeout(timeId.current), []);

  return timeId;
}
